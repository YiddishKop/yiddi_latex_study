\begin{tcolorbox}[mycard,title={\ProblemTitle}]
设实数列 $(a_n)$ 满足：$a_1=1$，且
\[
  a_{n+1} = a_n + \frac{1}{n a_n}\qquad (n\ge 1).
\]
(1) 证明 $a_n^2$ 单调增加；(2) 求证：存在常数 $C$ 使得 $a_n=\sqrt{2\ln n + C + o(1)}$；\\[2pt]
(3) 给出一个对 $a_{1000}$ 的三位小数近似。
\medskip

\textbf{作答提示}：可先考虑 $b_n=a_n^2$ 的增量；再用积分比较估计 $\sum \frac1{n a_n}$。
\end{tcolorbox}

% 答案卡片（可开关）
\ifnum\ShowAnswer=1
  \begin{tcolorbox}[mycard,title={参考解析（教师稿/发布时可隐藏）}]
  \textbf{(1)} 由递推式得
  $a_{n+1}^2-a_n^2=\dfrac{1}{n}+\dfrac{1}{n^2a_n^2}>0$，故单调增。\\
  \textbf{(2)} 设 $b_n=a_n^2$，则
  $b_{n+1}-b_n=\dfrac{1}{n}+\dfrac{1}{n^2 b_n}$，累加得
  $b_n=\ln n + \ln n + O(1)=2\ln n + C + o(1)$。\\
  \textbf{(3)} 代入数值迭代或用渐近式近似可得 $a_{1000}\approx \sqrt{2\ln 1000 + C}\ (\text{数值略})$。
  \end{tcolorbox}
\fi
